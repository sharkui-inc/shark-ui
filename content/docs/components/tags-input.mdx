## Usage

```tsx
import { Field, FieldLabel } from "@/components/ui/field";
import {
  TagsInput,
  TagsInputContext,
  TagsInputInput,
  TagsInputItem,
} from "@/components/ui/tags-input";
```

```tsx
<Field className="max-w-md">
  <FieldLabel>Frameworks</FieldLabel>
  <TagsInput defaultValue={["React", "Solid"]}>
    <TagsInputContext>
      {(api) => (
        <>
          {api.value.map((value, index) => (
            <TagsInputItem index={index} key={value} value={value}>
              {value}
            </TagsInputItem>
          ))}
          <TagsInputInput placeholder="Add framework" />
        </>
      )}
    </TagsInputContext>
  </TagsInput>
</Field>
```

## API Reference

### TagsInput

Root component. Renders a hidden input for form submission.

| Prop | Type | Default |
| --- | --- | --- |
| `size` | `"sm" \|| `onInputValueChange` | `(details) => void` | - |
| `showClear` | `boolean` | `true` || "lg"` | `"md"` |
| `defaultValue` | `string[]` | - |
| `value` | `string[]` | - |
| `inputValue` | `string` | - |
| `max` | `number` | `Infinity` |
| `maxLength` | `number` | - |
| `delimiter` | `string \| RegExp` | `","` |
| `allowDuplicates` | `boolean` | `false` |
| `allowOverflow` | `boolean` | - |
| `disabled` | `boolean` | - |
| `invalid` | `boolean` | - |
| `readOnly` | `boolean` | - |
| `editable` | `boolean` | `true` |
| `addOnPaste` | `boolean` | `false` |
| `blurBehavior` | `"add" \| "clear"` | - |
| `validate` | `(details) => boolean` | - |
| `sanitizeValue` | `(value: string) => string` | - |
| `onValueChange` | `(details) => void` | - |
| `onInputValueChange` | `(details) => void` | - |
| `showClear` | `boolean` | `true` ||| `onInputValueChange` | `(details) => void` | - |
| `showClear` | `boolean` | `true` ||| `onInputValueChange` | `(details) => void` | - |
| `showClear` | `boolean` | `true` |

### TagsInputInput

Borderless text field for adding new tags.

### TagsInputItem

Represents a single tag. Requires `index` and `value`. Pass the label as children. Renders the delete trigger and edit input. Chip height follows the root `size` (`sm`, `md`, `lg`). Supports `data-highlighted` when navigated by keyboard.

| Prop | Type | Default |
| --- | --- | --- |
| `showClear` | `boolean` | `true` |


### TagsInputClearTrigger

Clears all tags.
