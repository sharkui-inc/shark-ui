## Usage

```tsx
import { AspectRatio } from "@/components/ui/aspect-ratio";
```

```tsx
<AspectRatio>
  <img
    src="/image.jpg"
    alt="Description"
    className="object-cover size-full"
  />
</AspectRatio>
```
